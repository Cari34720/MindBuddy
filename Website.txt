    <script>
        document.addEventListener("DOMContentLoaded", function() {
            console.log("Mind Map Generator ready!");
            enableDragAndDrop();
            loadMindMap();
        });

        function uploadFile() {
            const fileInput = document.getElementById("fileInput");
            if (fileInput.files.length === 0) {
                alert("Bitte w채hle eine Datei aus.");
                return;
            }
            const file = fileInput.files[0];
            const reader = new FileReader();
            reader.onload = function(event) {
                generateMindMapFromText(event.target.result);
                saveMindMap();
            };
            reader.readAsText(file);
        }

        function generateMindMap() {
            const userInput = document.getElementById("textInput").value;
            if (userInput.trim() === "") {
                alert("Bitte gib ein Thema ein.");
                return;
            }
            generateMindMapFromText(userInput);
            saveMindMap();
        }

        function generateMindMapFromText(text) {
            const mindmapContainer = document.getElementById("mindmapContainer");
            mindmapContainer.innerHTML = "";
            const nodes = text.split(" ").map(word => word.trim()).filter(word => word !== "");
            const centerNode = document.createElement("div");
            centerNode.classList.add("node", "center");
            centerNode.textContent = nodes[0];
            centerNode.setAttribute("draggable", true);
            centerNode.addEventListener("dragstart", dragStart);
            mindmapContainer.appendChild(centerNode);
            nodes.slice(1).forEach(nodeText => {
                const node = document.createElement("div");
                node.classList.add("node");
                node.textContent = nodeText;
                node.setAttribute("draggable", true);
                node.addEventListener("dragstart", dragStart);
                mindmapContainer.appendChild(node);
            });
            saveMindMap();
        }

        function enableDragAndDrop() {
            document.addEventListener("dragover", function(event) {
                event.preventDefault();
            });
            document.addEventListener("drop", function(event) {
                event.preventDefault();
                const draggedNode = document.querySelector(".dragging");
                draggedNode.style.position = "absolute";
                draggedNode.style.left = `${event.clientX}px`;
                draggedNode.style.top = `${event.clientY}px`;
                saveMindMap();
            });
        }

        function dragStart(event) {
            event.target.classList.add("dragging");
            setTimeout(() => event.target.classList.remove("dragging"), 0);
        }

        function saveMindMap() {
            localStorage.setItem("mindmap", document.getElementById("mindmapContainer").innerHTML);
        }

        function loadMindMap() {
            const savedMindMap = localStorage.getItem("mindmap");
            if (savedMindMap) {
                document.getElementById("mindmapContainer").innerHTML = savedMindMap;
            }
        }

        }

        }
